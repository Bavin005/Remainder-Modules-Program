#include <stdio.h>
#include <string.h>

int main() {
    char str[100];
    int start = 0, maxLen = 1;
    int i, left, right;
    int len;

    printf("Enter a string: ");
    scanf("%s", str);

    len = strlen(str);

    for(i = 0; i < len; i++) {

        // Odd length palindrome
        left = i;
        right = i;

        while(left >= 0 && right < len && str[left] == str[right]) {
            if(right - left + 1 > maxLen) {
                start = left;
                maxLen = right - left + 1;
            }
            left--;
            right++;
        }

        // Even length palindrome
        left = i;
        right = i + 1;

        while(left >= 0 && right < len && str[left] == str[right]) {
            if(right - left + 1 > maxLen) {
                start = left;
                maxLen = right - left + 1;
            }
            left--;
            right++;
        }
    }

    printf("Longest Palindromic Substring: ");

    for(i = start; i < start + maxLen; i++) {
        printf("%c", str[i]);
    }

    printf("\n");

    return 0;
}
